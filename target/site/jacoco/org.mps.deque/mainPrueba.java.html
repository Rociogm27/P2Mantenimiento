package org.mps.deque;

public class mainPrueba {
        public static void main(String[] args){
            DoublyLinkedListDeque lista = new DoublyLinkedListDeque<>();
            System.out.println(lista.size());
            lista.append(3);
            System.out.println(lista.size());
            lista.prepend(2);
            System.out.println(lista.size());
            lista.append(4);
            System.out.println(lista.size());
            lista.append(9);
            System.out.println(lista.size());
            lista.mostrarLista();
            lista.deleteLast();
            System.out.println(lista.size());
            lista.deleteLast();
            System.out.println(lista.size());
            lista.mostrarLista();

        }
    }
